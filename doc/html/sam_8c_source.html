502int PhonemeStr2PhonemeIdx()
503{
504 unsigned char sign1;
505 unsigned char position = 0;
506 unsigned char srcpos = 0;
507
508 memset(stress, 0, 256); // Clear the stress table.
509
510 while((sign1 = input[srcpos]) != 155) { // 155 (\233) is end of line marker
511 signed int match;
512 unsigned char sign2 = input[++srcpos];
513 if ((match = full_match(sign1, sign2)) != -1) {
514 // Matched both characters (no wildcards)
515 phonemeindex[position++] = (unsigned char)match;
516 ++srcpos; // Skip the second character of the input as we've matched it
517 } else if ((match = wild_match(sign1)) != -1) {
518 // Matched just the first character (with second character matching '*'
519 phonemeindex[position++] = (unsigned char)match;
520 } else {
521 // Should be a stress character. Search through the
522 // stress table backwards.
523 match = 8; // End of stress table. FIXME: Don't hardcode.
524 while( (sign1 != stressInputTable[match]) && (match>0) ) --match;
525
526 if (match == 0) return 0; // failure
527
528 stress[position-1] = (unsigned char)match; // Set stress for prior phoneme
529 }
530 } //while
531
532 phonemeindex[position] = END;
533 return 1;

607// Rewrites the phonemes using the following rules:
608//
609// <DIPTHONG ENDING WITH WX> -> <DIPTHONG ENDING WITH WX> WX
610// <DIPTHONG NOT ENDING WITH WX> -> <DIPTHONG NOT ENDING WITH WX> YX
611// UL -> AX L
612// UM -> AX M
613// <STRESSED VOWEL> <SILENCE> <STRESSED VOWEL> -> <STRESSED VOWEL> <SILENCE> Q <VOWEL>
614// T R -> CH R
615// D R -> J R
616// <VOWEL> R -> <VOWEL> RX
617// <VOWEL> L -> <VOWEL> LX
618// G S -> G Z
619// K <VOWEL OR DIPTHONG NOT ENDING WITH IY> -> KX <VOWEL OR DIPTHONG NOT ENDING WITH IY>
620// G <VOWEL OR DIPTHONG NOT ENDING WITH IY> -> GX <VOWEL OR DIPTHONG NOT ENDING WITH IY>
621// S P -> S B
622// S T -> S D
623// S K -> S G
624// S KX -> S GX
625// <ALVEOLAR> UW -> <ALVEOLAR> UX
626// CH -> CH CH' (CH requires two phonemes to represent it)
627// J -> J J' (J requires two phonemes to represent it)
628// <UNSTRESSED VOWEL> T <PAUSE> -> <UNSTRESSED VOWEL> DX <PAUSE>
629// <UNSTRESSED VOWEL> D <PAUSE> -> <UNSTRESSED VOWEL> DX <PAUSE>

671void rule_dipthong(unsigned char p, unsigned short pf, unsigned char pos) {
672 // <DIPTHONG ENDING WITH WX> -> <DIPTHONG ENDING WITH WX> WX
673 // <DIPTHONG NOT ENDING WITH WX> -> <DIPTHONG NOT ENDING WITH WX> YX
674 // Example: OIL, COW
675
676 // If ends with IY, use YX, else use WX
677 unsigned char A = (pf & FLAG_DIP_YX) ? 21 : 20; // 'WX' = 20 'YX' = 21
678
679 // Insert at WX or YX following, copying the stress
680 if (A==20) drule("insert WX following dipthong NOT ending in IY sound");
681 else if (A==21) drule("insert YX following dipthong ending in IY sound");
682 Insert(pos+1, A, 0, stress[pos]);
683
684 if (p == 53) rule_alveolar_uw(pos); // Example: NEW, DEW, SUE, ZOO, THOO, TOO
685 else if (p == 42) rule_ch(pos); // Example: CHEW
686 else if (p == 44) rule_j(pos); // Example: JAY

689void PhonemeidxChangeAccordRuleA() {
690 unsigned char pos = 0; //mem66;
691 unsigned char p;
692
693 if (debug) printf("Parser2\n");
694
695 while((p = phonemeindex[pos]) != END) {
696 unsigned short pf;
697 unsigned char prior;
698
699 if (debug) printf("%d: %c%c\n", pos, phonemeNameTable1[p], PhonemeNameTable2[p]);
700
701 if (p == 0) { // Is phoneme pause?
702 ++pos;
703 continue;
704 }
705
706 pf = flags[p];
707 prior = phonemeindex[pos-1];
708
709 if ((pf & FLAG_DIPTHONG)) rule_dipthong(p, pf, pos);
710 else if (p == 78) ChangeRule(pos, 24, "UL -> AX L"); // Example: MEDDLE
711 else if (p == 79) ChangeRule(pos, 27, "UM -> AX M"); // Example: ASTRONOMY
712 else if (p == 80) ChangeRule(pos, 28, "UN -> AX N"); // Example: FUNCTION
713 else if ((pf & FLAG_VOWEL) && stress[pos]) {
714 // RULE:
715 // <STRESSED VOWEL> <SILENCE> <STRESSED VOWEL> -> <STRESSED VOWEL> <SILENCE> Q <VOWEL>
716 // EXAMPLE: AWAY EIGHT
717 if (!phonemeindex[pos+1]) { // If following phoneme is a pause, get next
718 p = phonemeindex[pos+2];
719 if (p!=END && (flags[p] & FLAG_VOWEL) && stress[pos+2]) {
720 drule("Insert glottal stop between two stressed vowels with space between them");
721 Insert(pos+2, 31, 0, 0); // 31 = 'Q'
722 }
723 }
724 } else if (p == pR) { // RULES FOR PHONEMES BEFORE R
725 if (prior == pT) change(pos-1,42, "T R -> CH R"); // Example: TRACK
726 else if (prior == pD) change(pos-1,44, "D R -> J R"); // Example: DRY
727 else if (flags[prior] & FLAG_VOWEL) change(pos, 18, "<VOWEL> R -> <VOWEL> RX"); // Example: ART
728 } else if (p == 24 && (flags[prior] & FLAG_VOWEL)) change(pos, 19, "<VOWEL> L -> <VOWEL> LX"); // Example: ALL
729 else if (prior == 60 && p == 32) { // 'G' 'S'
730 // Can't get to fire -
731 // 1. The G -> GX rule intervenes
732 // 2. Reciter already replaces GS -> GZ
733 change(pos, 38, "G S -> G Z");
734 } else if (p == 60) rule_g(pos);
735 else {
736 if (p == 72) { // 'K'
737 // K <VOWEL OR DIPTHONG NOT ENDING WITH IY> -> KX <VOWEL OR DIPTHONG NOT ENDING WITH IY>
738 // Example: COW
739 unsigned char Y = phonemeindex[pos+1];
740 // If at end, replace current phoneme with KX
741 if ((flags[Y] & FLAG_DIP_YX)==0 || Y==END) { // VOWELS AND DIPTHONGS ENDING WITH IY SOUND flag set?
742 change(pos, 75, "K <VOWEL OR DIPTHONG NOT ENDING WITH IY> -> KX <VOWEL OR DIPTHONG NOT ENDING WITH IY>");
743 p = 75;
744 pf = flags[p];
745 }
746 }
747
748 // Replace with softer version?
749 if ((flags[p] & FLAG_PLOSIVE) && (prior == 32)) { // 'S'
750 // RULE:
751 // S P -> S B
752 // S T -> S D
753 // S K -> S G
754 // S KX -> S GX
755 // Examples: SPY, STY, SKY, SCOWL
756
757 if (debug) printf("RULE: S* %c%c -> S* %c%c\n", phonemeNameTable1[p], PhonemeNameTable2[p],phonemeNameTable1[p-12], PhonemeNameTable2[p-12]);
758 phonemeindex[pos] = p-12;
759 } else if (!(pf & FLAG_PLOSIVE)) {
760 p = phonemeindex[pos];
761 if (p == 53) rule_alveolar_uw(pos); // Example: NEW, DEW, SUE, ZOO, THOO, TOO
762 else if (p == 42) rule_ch(pos); // Example: CHEW
763 else if (p == 44) rule_j(pos); // Example: JAY
764 }
765
766 if (p == 69 || p == 57) { // 'T', 'D'
767 // RULE: Soften T following vowel
768 // NOTE: This rule fails for cases such as "ODD"
769 // <UNSTRESSED VOWEL> T <PAUSE> -> <UNSTRESSED VOWEL> DX <PAUSE>
770 // <UNSTRESSED VOWEL> D <PAUSE> -> <UNSTRESSED VOWEL> DX <PAUSE>
771 // Example: PARTY, TARDY
772 if (flags[phonemeindex[pos-1]] & FLAG_VOWEL) {
773 p = phonemeindex[pos+1];
774 if (!p) p = phonemeindex[pos+2];
775 if ((flags[p] & FLAG_VOWEL) && !stress[pos+1]) change(pos,30, "Soften T or D following vowel or ER and preceding a pause -> DX");
776 }
777 }
778 }
779 pos++;
780 } // while
781}

785// Lengthen <FRICATIVE> or <VOICED> between <VOWEL> and <PUNCTUATION> by 1.5
786// <VOWEL> <RX | LX> <CONSONANT> - decrease <VOWEL> length by 1
787// <VOWEL> <UNVOICED PLOSIVE> - decrease vowel by 1/8th
788// <VOWEL> <UNVOICED CONSONANT> - increase vowel by 1/2 + 1
789// <NASAL> <STOP CONSONANT> - set nasal = 5, consonant = 6
790// <VOICED STOP CONSONANT> {optional silence} <STOP CONSONANT> - shorten both to 1/2 + 1
791// <LIQUID CONSONANT> <DIPTHONG> - decrease by 2
792//
793void AdjustLengthsAccordRuleb() {
794 // LENGTHEN VOWELS PRECEDING PUNCTUATION
795 //
796 // Search for punctuation. If found, back up to the first vowel, then
797 // process all phonemes between there and up to (but not including) the punctuation.
798 // If any phoneme is found that is a either a fricative or voiced, the duration is
799 // increased by (length * 1.5) + 1
800
801 // loop index
802 {
803 unsigned char X = 0;
804 unsigned char index;
805
806 while((index = phonemeindex[X]) != END) {
807 unsigned char loopIndex;
808
809 // not punctuation?
810 if((flags[index] & FLAG_PUNCT) == 0) {
811 ++X;
812 continue;
813 }
814
815 loopIndex = X;
816
817 while (--X && !(flags[phonemeindex[X]] & FLAG_VOWEL)); // back up while not a vowel
818 if (X == 0) break;
819
820 do {
821 // test for vowel
822 index = phonemeindex[X];
823
824 // test for fricative/unvoiced or not voiced
825 if(!(flags[index] & FLAG_FRICATIVE) || (flags[index] & FLAG_VOICED)) { //nochmal 黚erpr黤en
826 unsigned char A = phonemeLength[X];
827 // change phoneme length to (length * 1.5) + 1
828 drule_pre("Lengthen <FRICATIVE> or <VOICED> between <VOWEL> and <PUNCTUATION> by 1.5",X);
829 phonemeLength[X] = (A >> 1) + A + 1;
830 drule_post(X);
831 }
832 } while (++X != loopIndex);
833 X++;
834 } // while
835 }
836
837 // Similar to the above routine, but shorten vowels under some circumstances
838
839 // Loop through all phonemes
840 unsigned char loopIndex=0;
841 unsigned char index;
842
843 while((index = phonemeindex[loopIndex]) != END) {
844 unsigned char X = loopIndex;
845
846 if (flags[index] & FLAG_VOWEL) {
847 index = phonemeindex[loopIndex+1];
848 if (!(flags[index] & FLAG_CONSONANT)) {
849 if ((index == 18) || (index == 19)) { // 'RX', 'LX'
850 index = phonemeindex[loopIndex+2];
851 if ((flags[index] & FLAG_CONSONANT)) {
852 drule_pre("<VOWEL> <RX | LX> <CONSONANT> - decrease length of vowel by 1\n", loopIndex);
853 phonemeLength[loopIndex]--;
854 drule_post(loopIndex);
855 }
856 }
857 } else { // Got here if not <VOWEL>
858 unsigned short flag = (index == END) ? 65 : flags[index]; // 65 if end marker
859
860 if (!(flag & FLAG_VOICED)) { // Unvoiced
861 // *, .*, ?*, ,*, -*, DX, S*, SH, F*, TH, /H, /X, CH, P*, T*, K*, KX
862 if((flag & FLAG_PLOSIVE)) { // unvoiced plosive
863 // RULE: <VOWEL> <UNVOICED PLOSIVE>
864 // <VOWEL> <P*, T*, K*, KX>
865 drule_pre("<VOWEL> <UNVOICED PLOSIVE> - decrease vowel by 1/8th",loopIndex);
866 phonemeLength[loopIndex] -= (phonemeLength[loopIndex] >> 3);
867 drule_post(loopIndex);
868 }
869 } else {
870 unsigned char A;
871 drule_pre("<VOWEL> <VOICED CONSONANT> - increase vowel by 1/2 + 1\n",X-1);
872 // decrease length
873 A = phonemeLength[loopIndex];
874 phonemeLength[loopIndex] = (A >> 2) + A + 1; // 5/4*A + 1
875 drule_post(loopIndex);
876 }
877 }
878 } else if((flags[index] & FLAG_NASAL) != 0) { // nasal?
879 // RULE: <NASAL> <STOP CONSONANT>
880 // Set punctuation length to 6
881 // Set stop consonant length to 5
882 index = phonemeindex[++X];
883 if (index != END && (flags[index] & FLAG_STOPCONS)) {
884 drule("<NASAL> <STOP CONSONANT> - set nasal = 5, consonant = 6");
885 phonemeLength[X] = 6; // set stop consonant length to 6
886 phonemeLength[X-1] = 5; // set nasal length to 5
887 }
888 } else if((flags[index] & FLAG_STOPCONS)) { // (voiced) stop consonant?
889 // RULE: <VOICED STOP CONSONANT> {optional silence} <STOP CONSONANT>
890 // Shorten both to (length/2 + 1)
891
892 // move past silence
893 while ((index = phonemeindex[++X]) == 0);
894
895 if (index != END && (flags[index] & FLAG_STOPCONS)) {
896 // FIXME, this looks wrong?
897 // RULE: <UNVOICED STOP CONSONANT> {optional silence} <STOP CONSONANT>
898 drule("<UNVOICED STOP CONSONANT> {optional silence} <STOP CONSONANT> - shorten both to 1/2 + 1");
899 phonemeLength[X] = (phonemeLength[X] >> 1) + 1;
900 phonemeLength[loopIndex] = (phonemeLength[loopIndex] >> 1) + 1;
901 X = loopIndex;
902 }
903 } else if ((flags[index] & FLAG_LIQUIC)) { // liquic consonant?
904 // RULE: <VOICED NON-VOWEL> <DIPTHONG>
905 // Decrease <DIPTHONG> by 2
906 index = phonemeindex[X-1]; // prior phoneme;
907
908 // FIXME: The debug code here breaks the rule.
909 // prior phoneme a stop consonant>
910 if((flags[index] & FLAG_STOPCONS) != 0)
911 drule_pre("<LIQUID CONSONANT> <DIPTHONG> - decrease by 2",X);
912
913 phonemeLength[X] -= 2; // 20ms
914 drule_post(X);
915 }
916
917 ++loopIndex;
918 }
