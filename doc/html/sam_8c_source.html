264// The input[] buffer contains a string of phonemes and stress markers along
265// the lines of:
266//
267// DHAX KAET IHZ AH5GLIY. <0x9B>
268//
269// The byte 0x9B marks the end of the buffer. Some phonemes are 2 bytes
270// long, such as "DH" and "AX". Others are 1 byte long, such as "T" and "Z".
271// There are also stress markers, such as "5" and ".".
272//
273// The first character of the phonemes are stored in the table signInputTable1[].
274// The second character of the phonemes are stored in the table signInputTable2[].
275// The stress characters are arranged in low to high stress order in stressInputTable[].
276//
277// The following process is used to parse the input[] buffer:
278//
279// Repeat until the <0x9B> character is reached:
280//
281// First, a search is made for a 2 character match for phonemes that do not
282// end with the '*' (wildcard) character. On a match, the index of the phoneme
283// is added to phonemeIndex[] and the buffer position is advanced 2 bytes.
284//
285// If this fails, a search is made for a 1 character match against all
286// phoneme names ending with a '*' (wildcard). If this succeeds, the
287// phoneme is added to phonemeIndex[] and the buffer position is advanced
288// 1 byte.
289//
290// If this fails, search for a 1 character match in the stressInputTable[].
291// If this succeeds, the stress value is placed in the last stress[] table
292// at the same index of the last added phoneme, and the buffer position is
293// advanced by 1 byte.
294//
295// If this fails, return a 0.
296//
297// On success:
298//
299// 1. phonemeIndex[] will contain the index of all the phonemes.
300// 2. The last index in phonemeIndex[] will be 255.
301// 3. stress[] will contain the stress value for each phoneme
302
303// input[] holds the string of phonemes, each two bytes wide
304// signInputTable1[] holds the first character of each phoneme
305// signInputTable2[] holds te second character of each phoneme
306// phonemeIndex[] holds the indexes of the phonemes after parsing input[]
307//
308// The parser scans through the input[], finding the names of the phonemes
309// by searching signInputTable1[] and signInputTable2[]. On a match, it
310// copies the index of the phoneme into the phonemeIndexTable[].
311//
312// The character <0x9B> marks the end of text in input[]. When it is reached,
313// the index 255 is placed at the end of the phonemeIndexTable[], and the
314// function returns with a 1 indicating success.
315int Parser1()
316{
317 unsigned char sign1;
318 unsigned char position = 0;
319 unsigned char srcpos = 0;
320
321 memset(stress, 0, 256); // Clear the stress table.
322
323 while((sign1 = input[srcpos]) != 155) { // 155 (\233) is end of line marker
324 signed int match;
325 unsigned char sign2 = input[++srcpos];
326 if ((match = full_match(sign1, sign2)) != -1) {
327 // Matched both characters (no wildcards)
328 phonemeindex[position++] = (unsigned char)match;
329 ++srcpos; // Skip the second character of the input as we've matched it
330 } else if ((match = wild_match(sign1)) != -1) {
331 // Matched just the first character (with second character matching '*'
332 phonemeindex[position++] = (unsigned char)match;
333 } else {
334 // Should be a stress character. Search through the
335 // stress table backwards.
336 match = 8; // End of stress table. FIXME: Don't hardcode.
337 while( (sign1 != stressInputTable[match]) && (match>0) ) --match;
338
339 if (match == 0) return 0; // failure
340
341 stress[position-1] = (unsigned char)match; // Set stress for prior phoneme
342 }
343 } //while
344
345 phonemeindex[position] = END;
346 return 1;
347}

417// Rewrites the phonemes using the following rules:
418//
419// <DIPTHONG ENDING WITH WX> -> <DIPTHONG ENDING WITH WX> WX
420// <DIPTHONG NOT ENDING WITH WX> -> <DIPTHONG NOT ENDING WITH WX> YX
421// UL -> AX L
422// UM -> AX M
423// <STRESSED VOWEL> <SILENCE> <STRESSED VOWEL> -> <STRESSED VOWEL> <SILENCE> Q <VOWEL>
424// T R -> CH R
425// D R -> J R
426// <VOWEL> R -> <VOWEL> RX
427// <VOWEL> L -> <VOWEL> LX
428// G S -> G Z
429// K <VOWEL OR DIPTHONG NOT ENDING WITH IY> -> KX <VOWEL OR DIPTHONG NOT ENDING WITH IY>
430// G <VOWEL OR DIPTHONG NOT ENDING WITH IY> -> GX <VOWEL OR DIPTHONG NOT ENDING WITH IY>
431// S P -> S B
432// S T -> S D
433// S K -> S G
434// S KX -> S GX
435// <ALVEOLAR> UW -> <ALVEOLAR> UX
436// CH -> CH CH' (CH requires two phonemes to represent it)
437// J -> J J' (J requires two phonemes to represent it)
438// <UNSTRESSED VOWEL> T <PAUSE> -> <UNSTRESSED VOWEL> DX <PAUSE>
439// <UNSTRESSED VOWEL> D <PAUSE> -> <UNSTRESSED VOWEL> DX <PAUSE>

481void rule_dipthong(unsigned char p, unsigned short pf, unsigned char pos) {
482 // <DIPTHONG ENDING WITH WX> -> <DIPTHONG ENDING WITH WX> WX
483 // <DIPTHONG NOT ENDING WITH WX> -> <DIPTHONG NOT ENDING WITH WX> YX
484 // Example: OIL, COW
485
486 // If ends with IY, use YX, else use WX
487 unsigned char A = (pf & FLAG_DIP_YX) ? 21 : 20; // 'WX' = 20 'YX' = 21
488
489 // Insert at WX or YX following, copying the stress
490 if (A==20) drule("insert WX following dipthong NOT ending in IY sound");
491 else if (A==21) drule("insert YX following dipthong ending in IY sound");
492 Insert(pos+1, A, 0, stress[pos]);
493
494 if (p == 53) rule_alveolar_uw(pos); // Example: NEW, DEW, SUE, ZOO, THOO, TOO
495 else if (p == 42) rule_ch(pos); // Example: CHEW
496 else if (p == 44) rule_j(pos); // Example: JAY
497}

499void Parser2() {
500 unsigned char pos = 0; //mem66;
501 unsigned char p;
502
503 if (debug) printf("Parser2\n");
504
505 while((p = phonemeindex[pos]) != END) {
506 unsigned short pf;
507 unsigned char prior;
508
509 if (debug) printf("%d: %c%c\n", pos, signInputTable1[p], signInputTable2[p]);
510
511 if (p == 0) { // Is phoneme pause?
512 ++pos;
513 continue;
514 }
515
516 pf = flags[p];
517 prior = phonemeindex[pos-1];
518
519 if ((pf & FLAG_DIPTHONG)) rule_dipthong(p, pf, pos);
520 else if (p == 78) ChangeRule(pos, 24, "UL -> AX L"); // Example: MEDDLE
521 else if (p == 79) ChangeRule(pos, 27, "UM -> AX M"); // Example: ASTRONOMY
522 else if (p == 80) ChangeRule(pos, 28, "UN -> AX N"); // Example: FUNCTION
523 else if ((pf & FLAG_VOWEL) && stress[pos]) {
524 // RULE:
525 // <STRESSED VOWEL> <SILENCE> <STRESSED VOWEL> -> <STRESSED VOWEL> <SILENCE> Q <VOWEL>
526 // EXAMPLE: AWAY EIGHT
527 if (!phonemeindex[pos+1]) { // If following phoneme is a pause, get next
528 p = phonemeindex[pos+2];
529 if (p!=END && (flags[p] & FLAG_VOWEL) && stress[pos+2]) {
530 drule("Insert glottal stop between two stressed vowels with space between them");
531 Insert(pos+2, 31, 0, 0); // 31 = 'Q'
532 }
533 }
534 } else if (p == pR) { // RULES FOR PHONEMES BEFORE R
535 if (prior == pT) change(pos-1,42, "T R -> CH R"); // Example: TRACK
536 else if (prior == pD) change(pos-1,44, "D R -> J R"); // Example: DRY
537 else if (flags[prior] & FLAG_VOWEL) change(pos, 18, "<VOWEL> R -> <VOWEL> RX"); // Example: ART
538 } else if (p == 24 && (flags[prior] & FLAG_VOWEL)) change(pos, 19, "<VOWEL> L -> <VOWEL> LX"); // Example: ALL
539 else if (prior == 60 && p == 32) { // 'G' 'S'
540 // Can't get to fire -
541 // 1. The G -> GX rule intervenes
542 // 2. Reciter already replaces GS -> GZ
543 change(pos, 38, "G S -> G Z");
544 } else if (p == 60) rule_g(pos);
545 else {
546 if (p == 72) { // 'K'
547 // K <VOWEL OR DIPTHONG NOT ENDING WITH IY> -> KX <VOWEL OR DIPTHONG NOT ENDING WITH IY>
548 // Example: COW
549 unsigned char Y = phonemeindex[pos+1];
550 // If at end, replace current phoneme with KX
551 if ((flags[Y] & FLAG_DIP_YX)==0 || Y==END) { // VOWELS AND DIPTHONGS ENDING WITH IY SOUND flag set?
552 change(pos, 75, "K <VOWEL OR DIPTHONG NOT ENDING WITH IY> -> KX <VOWEL OR DIPTHONG NOT ENDING WITH IY>");
553 p = 75;
554 pf = flags[p];
555 }
556 }
557
558 // Replace with softer version?
559 if ((flags[p] & FLAG_PLOSIVE) && (prior == 32)) { // 'S'
560 // RULE:
561 // S P -> S B
562 // S T -> S D
563 // S K -> S G
564 // S KX -> S GX
565 // Examples: SPY, STY, SKY, SCOWL
566
567 if (debug) printf("RULE: S* %c%c -> S* %c%c\n", signInputTable1[p], signInputTable2[p],signInputTable1[p-12], signInputTable2[p-12]);
568 phonemeindex[pos] = p-12;
569 } else if (!(pf & FLAG_PLOSIVE)) {
570 p = phonemeindex[pos];
571 if (p == 53) rule_alveolar_uw(pos); // Example: NEW, DEW, SUE, ZOO, THOO, TOO
572 else if (p == 42) rule_ch(pos); // Example: CHEW
573 else if (p == 44) rule_j(pos); // Example: JAY
574 }
575
576 if (p == 69 || p == 57) { // 'T', 'D'
577 // RULE: Soften T following vowel
578 // NOTE: This rule fails for cases such as "ODD"
579 // <UNSTRESSED VOWEL> T <PAUSE> -> <UNSTRESSED VOWEL> DX <PAUSE>
580 // <UNSTRESSED VOWEL> D <PAUSE> -> <UNSTRESSED VOWEL> DX <PAUSE>
581 // Example: PARTY, TARDY
582 if (flags[phonemeindex[pos-1]] & FLAG_VOWEL) {
583 p = phonemeindex[pos+1];
584 if (!p) p = phonemeindex[pos+2];
585 if ((flags[p] & FLAG_VOWEL) && !stress[pos+1]) change(pos,30, "Soften T or D following vowel or ER and preceding a pause -> DX");
586 }
587 }
588 }
589 pos++;
590 } // while
591}

593// Applies various rules that adjust the lengths of phonemes
594//
595// Lengthen <FRICATIVE> or <VOICED> between <VOWEL> and <PUNCTUATION> by 1.5
596// <VOWEL> <RX | LX> <CONSONANT> - decrease <VOWEL> length by 1
597// <VOWEL> <UNVOICED PLOSIVE> - decrease vowel by 1/8th
598// <VOWEL> <UNVOICED CONSONANT> - increase vowel by 1/2 + 1
599// <NASAL> <STOP CONSONANT> - set nasal = 5, consonant = 6
600// <VOICED STOP CONSONANT> {optional silence} <STOP CONSONANT> - shorten both to 1/2 + 1
601// <LIQUID CONSONANT> <DIPTHONG> - decrease by 2
602//
603void AdjustLengths() {
604 // LENGTHEN VOWELS PRECEDING PUNCTUATION
605 //
606 // Search for punctuation. If found, back up to the first vowel, then
607 // process all phonemes between there and up to (but not including) the punctuation.
608 // If any phoneme is found that is a either a fricative or voiced, the duration is
609 // increased by (length * 1.5) + 1
610
611 // loop index
612 {
613 unsigned char X = 0;
614 unsigned char index;
615
616 while((index = phonemeindex[X]) != END) {
617 unsigned char loopIndex;
618
619 // not punctuation?
620 if((flags[index] & FLAG_PUNCT) == 0) {
621 ++X;
622 continue;
623 }
624
625 loopIndex = X;
626
627 while (--X && !(flags[phonemeindex[X]] & FLAG_VOWEL)); // back up while not a vowel
628 if (X == 0) break;
629
630 do {
631 // test for vowel
632 index = phonemeindex[X];
633
634 // test for fricative/unvoiced or not voiced
635 if(!(flags[index] & FLAG_FRICATIVE) || (flags[index] & FLAG_VOICED)) { //nochmal berprfen
636 unsigned char A = phonemeLength[X];
637 // change phoneme length to (length * 1.5) + 1
638 drule_pre("Lengthen <FRICATIVE> or <VOICED> between <VOWEL> and <PUNCTUATION> by 1.5",X);
639 phonemeLength[X] = (A >> 1) + A + 1;
640 drule_post(X);
641 }
642 } while (++X != loopIndex);
643 X++;
644 } // while
645 }
