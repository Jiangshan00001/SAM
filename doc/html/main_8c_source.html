29void WriteWav(char* filename, char* buffer, int bufferlength)
30{
31 unsigned int filesize;
32 unsigned int fmtlength = 16;
33 unsigned short int format=1; //PCM
34 unsigned short int channels=1;
35 unsigned int samplerate = 22050;
36 unsigned short int blockalign = 1;
37 unsigned short int bitspersample=8;
38
39 FILE *file;
40 fopen_s(&file, filename, "wb");
41 if (file == NULL) return;
42 //RIFF header
43 fwrite("RIFF", 4, 1,file);
44 filesize=bufferlength + 12 + 16 + 8 - 8;
45 fwrite(&filesize, 4, 1, file);
46 fwrite("WAVE", 4, 1, file);
47
48 //format chunk
49 fwrite("fmt ", 4, 1, file);
50 fwrite(&fmtlength, 4, 1, file);
51 fwrite(&format, 2, 1, file);
52 fwrite(&channels, 2, 1, file);
53 fwrite(&samplerate, 4, 1, file);
54 fwrite(&samplerate, 4, 1, file); // bytes/second
55 fwrite(&blockalign, 2, 1, file);
56 fwrite(&bitspersample, 2, 1, file);
57
58 //data chunk
59 fwrite("data", 4, 1, file);
60 fwrite(&bufferlength, 4, 1, file);
61 fwrite(buffer, bufferlength, 1, file);
62
63 fclose(file);
64}
