105// -------------------------------------------------------------------------
106//Code48227
107// Render a sampled sound from the sampleTable.
108//
109// Phoneme Sample Start Sample End
110// 32: S* 15 255
111// 33: SH 257 511
112// 34: F* 559 767
113// 35: TH 583 767
114// 36: /H 903 1023
115// 37: /X 1135 1279
116// 38: Z* 84 119
117// 39: ZH 340 375
118// 40: V* 596 639
119// 41: DH 596 631
120//
121// 42: CH
122// 43: ** 399 511
123//
124// 44: J*
125// 45: ** 257 276
126// 46: **
127//
128// 66: P*
129// 67: ** 743 767
130// 68: **
131//
132// 69: T*
133// 70: ** 231 255
134// 71: **
135//
136// The SampledPhonemesTable[] holds flags indicating if a phoneme is
137// voiced or not. If the upper 5 bits are zero, the sample is voiced.
138//
139// Samples in the sampleTable are compressed, with bits being converted to
140// bytes from high bit to low, as follows:
141//
142// unvoiced 0 bit -> X
143// unvoiced 1 bit -> 5
144//
145// voiced 0 bit -> 6
146// voiced 1 bit -> 24
147//
148// Where X is a value from the table:
149//
150// { 0x18, 0x1A, 0x17, 0x17, 0x17 };
151//
152// The index into this table is determined by masking off the lower
153// 3 bits from the SampledPhonemesTable:
154//
155// index = (SampledPhonemesTable[i] & 7) - 1;
156//
157// For voices samples, samples are interleaved between voiced output.
158
159
160void RenderSample(unsigned char *mem66, unsigned char consonantFlag, unsigned char mem49)
161{
162 // mem49 == current phoneme's index
163
164 // mask low three bits and subtract 1 get value to
165 // convert 0 bits on unvoiced samples.
166 unsigned char hibyte = (consonantFlag & 7)-1;
167
168 // determine which offset to use from table { 0x18, 0x1A, 0x17, 0x17, 0x17 }
169 // T, S, Z 0 0x18
170 // CH, J, SH, ZH 1 0x1A
171 // P, F*, V, TH, DH 2 0x17
172 // /H 3 0x17
173 // /X 4 0x17
174
175 unsigned short hi = hibyte*256;
176 // voiced sample?
177 unsigned char pitchl = consonantFlag & 248;
178 if(pitchl == 0) {
179 // voiced phoneme: Z*, ZH, V*, DH
180 pitchl = pitches[mem49] >> 4;
181 *mem66 = RenderVoicedSample(hi, *mem66, pitchl ^ 255);
182 }
183 else
184 RenderUnvoicedSample(hi, pitchl^255, tab48426[hibyte]);
185}

189// CREATE FRAMES
190//
191// The length parameter in the list corresponds to the number of frames
192// to expand the phoneme to. Each frame represents 10 milliseconds of time.
193// So a phoneme with a length of 7 = 7 frames = 70 milliseconds duration.
194//
195// The parameters are copied from the phoneme to the frame verbatim.
196//
197static void CreateFrames()
198{
199 unsigned char X = 0;
200 unsigned int i = 0;
201 while(i < 256) {
202 // get the phoneme at the index
203 unsigned char phoneme = phonemeIndexOutput[i];
204 unsigned char phase1;
205 unsigned phase2;
206
207 // if terminal phoneme, exit the loop
208 if (phoneme == 255) break;
209
210 if (phoneme == PHONEME_PERIOD) AddInflection(RISING_INFLECTION, X);
211 else if (phoneme == PHONEME_QUESTION) AddInflection(FALLING_INFLECTION, X);
212
213 // get the stress amount (more stress = higher pitch)
214 phase1 = tab47492[stressOutput[i] + 1];
215
216 // get number of frames to write
217 phase2 = phonemeLengthOutput[i];
218
219 // copy from the source to the frames list
220 do {
221 frequency1[X] = freq1data[phoneme]; // F1 frequency
222 frequency2[X] = freq2data[phoneme]; // F2 frequency
223 frequency3[X] = freq3data[phoneme]; // F3 frequency
224 amplitude1[X] = ampl1data[phoneme]; // F1 amplitude
225 amplitude2[X] = ampl2data[phoneme]; // F2 amplitude
226 amplitude3[X] = ampl3data[phoneme]; // F3 amplitude
227 sampledConsonantFlag[X] = sampledConsonantFlags[phoneme]; // phoneme data for sampled consonants
228 pitches[X] = pitch + phase1; // pitch
229 ++X;
230 } while(--phase2 != 0);
231
232 ++i;
233 }
234}

271// RENDER THE PHONEMES IN THE LIST
272//
273// The phoneme list is converted into sound through the steps:
274//
275// 1. Copy each phoneme <length> number of times into the frames list,
276// where each frame represents 10 milliseconds of sound.
277//
278// 2. Determine the transitions lengths between phonemes, and linearly
279// interpolate the values across the frames.
280//
281// 3. Offset the pitches by the fundamental frequency.
282//
283// 4. Render the each frame.
284void Render()
285{
286 unsigned char t;
287
288 if (phonemeIndexOutput[0] == 255) return; //exit if no data
289
290 CreateFrames();
291 t = CreateTransitions();
292
293 if (!singmode) AssignPitchContour();
294 RescaleAmplitude();
295
296 if (debug) {
297 PrintOutput(sampledConsonantFlag, frequency1, frequency2, frequency3, amplitude1, amplitude2, amplitude3, pitches);
298 }
299
300 ProcessFrames(t);
301}

332/*
333 SAM's voice can be altered by changing the frequencies of the
334 mouth formant (F1) and the throat formant (F2). Only the voiced
335 phonemes (5-29 and 48-53) are altered.
336*/
337void SetMouthThroat(unsigned char mouth, unsigned char throat)
338{
339 // mouth formants (F1) 5..29
340 static const unsigned char mouthFormants5_29[30] = {
341 0, 0, 0, 0, 0, 10,
342 14, 19, 24, 27, 23, 21, 16, 20, 14, 18, 14, 18, 18,
343 16, 13, 15, 11, 18, 14, 11, 9, 6, 6, 6};
344
345 // throat formants (F2) 5..29
346 static const unsigned char throatFormants5_29[30] = {
347 255, 255,
348 255, 255, 255, 84, 73, 67, 63, 40, 44, 31, 37, 45, 73, 49,
349 36, 30, 51, 37, 29, 69, 24, 50, 30, 24, 83, 46, 54, 86,
350 };
351
352 // there must be no zeros in this 2 tables
353 // formant 1 frequencies (mouth) 48..53
354 static const unsigned char mouthFormants48_53[6] = {19, 27, 21, 27, 18, 13};
355
356 // formant 2 frequencies (throat) 48..53
357 static const unsigned char throatFormants48_53[6] = {72, 39, 31, 43, 30, 34};
358
359 unsigned char newFrequency = 0;
360 unsigned char pos = 5;
361
362 // recalculate formant frequencies 5..29 for the mouth (F1) and throat (F2)
363 while(pos < 30)
364 {
365 // recalculate mouth frequency
366 unsigned char initialFrequency = mouthFormants5_29[pos];
367 if (initialFrequency != 0) newFrequency = trans(mouth, initialFrequency);
368 freq1data[pos] = newFrequency;
369
370 // recalculate throat frequency
371 initialFrequency = throatFormants5_29[pos];
372 if(initialFrequency != 0) newFrequency = trans(throat, initialFrequency);
373 freq2data[pos] = newFrequency;
374 pos++;
375 }
376
377 // recalculate formant frequencies 48..53
378 pos = 0;
379 while(pos < 6) {
380 // recalculate F1 (mouth formant)
381 unsigned char initialFrequency = mouthFormants48_53[pos];
382 freq1data[pos+48] = trans(mouth, initialFrequency);
383
384 // recalculate F2 (throat formant)
385 initialFrequency = throatFormants48_53[pos];
386 freq2data[pos+48] = trans(throat, initialFrequency);
387 pos++;
388 }
