132//Code48227
133// Render a sampled sound from the sampleTable.
134//
135// Phoneme Sample Start Sample End
136// 32: S* 15 255
137// 33: SH 257 511
138// 34: F* 559 767
139// 35: TH 583 767
140// 36: /H 903 1023
141// 37: /X 1135 1279
142// 38: Z* 84 119
143// 39: ZH 340 375
144// 40: V* 596 639
145// 41: DH 596 631
146//
147// 42: CH
148// 43: ** 399 511
149//
150// 44: J*
151// 45: ** 257 276
152// 46: **
153//
154// 66: P*
155// 67: ** 743 767
156// 68: **
157//
158// 69: T*
159// 70: ** 231 255
160// 71: **
161//
162// The SampledPhonemesTable[] holds flags indicating if a phoneme is
163// voiced or not. If the upper 5 bits are zero, the sample is voiced.
164//
165// Samples in the sampleTable are compressed, with bits being converted to
166// bytes from high bit to low, as follows:
167//
168// unvoiced 0 bit -> X
169// unvoiced 1 bit -> 5
170//
171// voiced 0 bit -> 6
172// voiced 1 bit -> 24
173//
174// Where X is a value from the table:
175//
176// { 0x18, 0x1A, 0x17, 0x17, 0x17 };
177//
178// The index into this table is determined by masking off the lower
179// 3 bits from the SampledPhonemesTable:
180//
181// index = (SampledPhonemesTable[i] & 7) - 1;
182//
183// For voices samples, samples are interleaved between voiced output.
184
185
186void RenderConsonantSample(unsigned char *offset, unsigned char consonantFlag, unsigned char frame_index)
187{
188 // mem49 == current phoneme's index
189
190 // mask low three bits and subtract 1 get value to
191 // convert 0 bits on unvoiced samples.
192 unsigned char hibyte = (consonantFlag & 7)-1;
193
194 // determine which offset to use from table { 0x18, 0x1A, 0x17, 0x17, 0x17 }
195 // T, S, Z 0 0x18
196 // CH, J, SH, ZH 1 0x1A
197 // P, F*, V, TH, DH 2 0x17
198 // /H 3 0x17
199 // /X 4 0x17
200
201 unsigned short hi = hibyte*0x100;
202 // voiced sample?
203 unsigned char pitchl = consonantFlag & 0xf8;
204 if(pitchl == 0) {
205 // voiced phoneme: Z*, ZH, V*, DH
206 pitchl = pitches[frame_index] >> 4;
207 *offset = RenderVoicedSample(hi, *offset, 255-pitchl);
208 }
209 else
210 RenderUnvoicedSample(hi, 255-pitchl, consonantAmp[hibyte]);
211}

215// CREATE FRAMES
216//
217// The length parameter in the list corresponds to the number of frames
218// to expand the phoneme to. Each frame represents 10 milliseconds of time.
219// So a phoneme with a length of 7 = 7 frames = 70 milliseconds duration.
220//
221// The parameters are copied from the phoneme to the frame verbatim.
222//
223void CreateFrames()
224{
225 unsigned char X = 0;
226 unsigned int i = 0;
227 while(i < 256) {
228 // get the phoneme at the index
229 unsigned char phoneme = phonemeIndexOutput[i];
230 unsigned char phase1;
231 unsigned plens;
232
233 // if terminal phoneme, exit the loop
234 if (phoneme == 255) break;
235
236 if (phoneme == PHONEME_PERIOD) AddInflection(RISING_INFLECTION, X);
237 else if (phoneme == PHONEME_QUESTION) AddInflection(FALLING_INFLECTION, X);
238
239 // get the stress amount (more stress = higher pitch)
240 phase1 = stressToPitchOffset[stressOutput[i] + 1];
241
242 // get number of frames to write
243 plens = phonemeLengthOutput[i];
244 if (plens>0){
245 // copy from the source to the frames list
246 do {
247 frequency1[X] = freq1data[phoneme]; // F1 frequency
248 frequency2[X] = freq2data[phoneme]; // F2 frequency
249 frequency3[X] = freq3data[phoneme]; // F3 frequency
250 amplitude1[X] = ampl1data[phoneme]; // F1 amplitude
251 amplitude2[X] = ampl2data[phoneme]; // F2 amplitude
252 amplitude3[X] = ampl3data[phoneme]; // F3 amplitude
253 sampledConsonantFlag[X] = sampledConsonantFlags[phoneme]; // phoneme data for sampled consonants
254 pitches[X] = pitch + phase1; // pitch
255 ++X;
256 } while(--plens != 0);
257 }
258
259 ++i;
260 }
261}
